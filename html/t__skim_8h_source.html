368  int UpdateCover(int seed, std::unordered_map<int, int>* influence_change, const std::vector<int>& covered_nodes) {
369  int num_covered_nodes = 0;
370  for (int j=0; j < covered_nodes.size(); j++) {
371  int covered_node = covered_nodes[j];
372  if (cover_->IsCovered(covered_node)) {
373  // Covered by a different node
374  continue;
375  }
376  LOG_M(DEBUG3, "Seed=" << seed << "Covered node= " << covered_node);
377  ++num_covered_nodes;
378  cover_->AddNodeToSeed(seed, covered_node);
379  node_influence_.erase(covered_node);
380  }
381 
382  for (int j=0; j < covered_nodes.size(); j++) {
383  int covered_node = covered_nodes[j];
384  if (reachable_nodes_.count(covered_node) == 0) {
385  continue;
386  }
387  for (int i=0; i < reachable_nodes_[covered_node].size(); i++) {
388  int reachable_node_from_covered_node = reachable_nodes_[covered_node][i];
389  if (cover_->IsCovered(reachable_node_from_covered_node)) {
390  // Covered by a different node
391  continue;
392  }
393  LOG_M(DEBUG3, "Node " << reachable_node_from_covered_node << " reachable from " << covered_node);
394  if (influence_change->count(reachable_node_from_covered_node) == 0) {
395  (*influence_change)[reachable_node_from_covered_node] = node_influence_[reachable_node_from_covered_node];
396  }
397  node_influence_[reachable_node_from_covered_node] = node_influence_[reachable_node_from_covered_node] - 1;
398  }
399  reachable_nodes_.erase(covered_node);
400  }
401  LOG_M(DEBUG3, "Seed=" << seed << " num actual covered nodes= " << num_covered_nodes);
402  return num_covered_nodes;
403  }

430  int Run(bool ShouldRunPreRunInit = true) {
431  if (ShouldRunPreRunInit) {
432  PreRunInit();
433  }
434  int num_covered_nodes = 0;
435  // Itrating the permutation of nodes
436  int num_passed = 0;
437  for (int i=0; i < rankers_nodes_.size(); i++) {
438  int source_node_id = rankers_nodes_[i];
439  // std::cout << "\r" << num_passed << "/" << rankers_nodes_.size();
440  ++num_passed;
441  if (cover_->IsCovered(source_node_id)) {
442  continue;
443  }
444  LOG_M(DEBUG3, "Iterating permutation, current node = " << source_node_id);
445  std::vector<int> current_iteration_seed_set;
446  const std::vector<int>& visited_nodes = CalculateVisitedNodes(source_node_id);
447  for (int i=0; i < visited_nodes.size(); i++) {
448  int visited_node = visited_nodes[i];
449  LOG_M(DEBUG3, "visited node = " << visited_node);
450  if (cover_->IsCovered(visited_node)) {
451  continue;
452  }
453  UpdateInfluceAndSeedSet(source_node_id, visited_node, &current_iteration_seed_set);
454  }
455 
456  for (int i=0; i <current_iteration_seed_set.size(); i++) {
457  int seed = current_iteration_seed_set[i];
458  if (cover_->IsCovered(source_node_id) || node_influence_[seed] < min_influence_for_seed_) {
459  continue;
460  }
461  std::unordered_map<int, int> influence_change;
462  LOG_M(DEBUG5, "Adding Seed node =" << seed );
463  double estimate_seed_cover_size = ( min_influence_for_seed_ * (static_cast<double>(rankers_nodes_.size()) / static_cast<double>(num_passed)) );
464  cover_->SetSeedEstimate(seed, estimate_seed_cover_size);
465  num_covered_nodes += AddSeed(seed, &influence_change);
466  }
467  }
468 
469  LOG_M(DEBUG3, " num covered nodes =" << num_covered_nodes <<
470  " num nodes left =" << node_influence_.size());
471  std::set< std::pair<int, int> , cmp_pair> heap;
472  for (auto it = node_influence_.begin(); it != node_influence_.end(); it++) {
473  LOG_M(DEBUG3, "Node influence raw node =" << it->first << " Influence=" << it->second);
474  heap.insert(std::make_pair(it->first, it->second));
475  }
476  int last_cover_size = INT_MAX;
477  int last_influence = INT_MAX;
478  while (!heap.empty()) {
479  auto node_details = (*heap.begin());
480  heap.erase(heap.begin());
481  int seed = node_details.first;
482  int seed_influence = node_details.second;
483  if (seed_influence > last_influence) {
484  LOG_M(NOTICE, "Assumption failed num influence increased" <<
485  "last influence=" << last_influence <<
486  "current influence=" << seed_influence);
487  assert(seed_influence <= last_influence);
488  }
489  last_influence = seed_influence;
490  if (cover_->IsCovered(seed)) {
491  assert(node_influence_.count(seed) == 0);
492  continue;
493  }
494  std::unordered_map<int, int> influence_change;
495  auto num_covered_nodes_by_seed = AddSeed(seed, &influence_change);
496  if (this->wanted_nodes_.size() == this->graph_->GetNumNodes()) {
497  assert(num_covered_nodes_by_seed == seed_influence);
498  }
499  if (num_covered_nodes_by_seed > last_cover_size &&
500  this->wanted_nodes_.size() == this->graph_->GetNumNodes()) {
501  LOG_M(NOTICE, "Assumption failed num covered nodes increased" <<
502  " last=" << last_cover_size <<
503  " current="<< num_covered_nodes_by_seed <<
504  " seed=" << seed <<
505  " seed influence=" << seed_influence);
506  assert(num_covered_nodes_by_seed <= last_cover_size);
507  }
508  last_cover_size = num_covered_nodes_by_seed;
509  LOG_M(DEBUG3, "Adding Seed node =" << seed << " Influence=" << seed_influence << "covered nodes=" << num_covered_nodes_by_seed);
510  num_covered_nodes += num_covered_nodes_by_seed;
511  cover_->SetSeedEstimate(seed, num_covered_nodes_by_seed);
512  for (std::unordered_map<int, int>::iterator change_it = influence_change.begin();
513  change_it != influence_change.end();
514  change_it++) {
515  heap.erase(std::make_pair(change_it->first, change_it->second));
516  heap.insert( std::make_pair(change_it->first, node_influence_[change_it->first]));
517  }
518  }
519  return 0;
520  }
