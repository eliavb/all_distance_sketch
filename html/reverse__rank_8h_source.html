217 static void CalculateReverseRank(int source_node_id,
218  graph::Graph<T> * graph,
219  GraphSketch * graph_sketch,
220  std::vector<int> * ranking,
221  CallBacks* call_backs) {
222  int max_node_id = graph->GetMxNId();
223  ranking->clear();
224  ranking->resize(max_node_id, constants::UNREACHABLE);
225  (*ranking)[source_node_id] = 0;
226  TBitSet poped; poped.resize(max_node_id);
227  TBitSet touced; touced.resize(max_node_id);
228  std::vector<graph::EdgeWeight> min_distance;
229  min_distance.resize(max_node_id, constants::UNREACHABLE);
230  min_distance[source_node_id] = 0;
231  std::set< RankData, compareRankNode > heap;
232  touced[source_node_id] = true;
233  heap.insert(RankData(source_node_id, (*ranking)[source_node_id], 0));
234 
235  LOG_M(DEBUG3, "Starting Reverse rank from node=" << source_node_id);
236  call_backs->Started(source_node_id, graph);
237  double rank_last = 0;
238  while (!heap.empty()) {
239  // Get the node with the minimum rank
240  int visited_node_id = heap.begin()->node_id;
241  auto node_data = *heap.begin();
242  graph::EdgeWeight distance_from_source_to_visited_node = node_data.distance;
243  (*ranking)[visited_node_id] = node_data.rank;
244  call_backs->NodePopedFromHeap(visited_node_id, node_data);
245  LOG_M(DEBUG3, "Poped node=" << visited_node_id << " Distance=" << node_data.distance <<
246  " Rank=" << node_data.rank);
247  assert(rank_last <= node_data.rank);
248  rank_last = node_data.rank;
249  _unused(rank_last);
250  heap.erase(heap.begin());
251  poped[visited_node_id] = true;
252  if (call_backs->ShouldPrune(visited_node_id, node_data)) {
253  continue;
254  }
255 
256  if (call_backs->ShouldStop()) {
257  return;
258  }
259 
260  // Visit each edge exiting visited_node_id
261  graph::TEdgesWeights * nodeWeights = graph->GetNodeWeights(visited_node_id);
262  typename T::TNodeI neighbors = graph->GetNI(visited_node_id);
263  for (int i = 0 ; i < neighbors.GetOutDeg(); i++) {
264  int id_of_neighbor_of_visited_node = neighbors.GetOutNId(i);
265  LOG_M(DEBUG3, "neighbors node=" << id_of_neighbor_of_visited_node);
266  if (poped[id_of_neighbor_of_visited_node]) {
267  continue;
268  }
269  std::pair<bool, graph::EdgeWeight> edge_u_v = graph->GetEdgeWeight(visited_node_id, id_of_neighbor_of_visited_node);
270  graph::EdgeWeight distance_through_u = distance_from_source_to_visited_node + edge_u_v.second; // (*nodeWeights)[i].weight;
271  LOG_M(DEBUG3, "edge weight between " <<
272  " visited_node_id = " << visited_node_id <<
273  " id_of_neighbor_of_visited_node = " << id_of_neighbor_of_visited_node <<
274  " edge weight = " << (*nodeWeights)[i].get_edge_weight() <<
275  " Distance before relax=" << distance_from_source_to_visited_node <<
276  " distance through visited_node_id=" << distance_through_u);
277 
278  if (touced[id_of_neighbor_of_visited_node] == false || distance_through_u < min_distance[id_of_neighbor_of_visited_node]) {
279  int rankOfSourceNode = EstimateReverseRankUpperBound<T>(graph,
280  graph_sketch,
281  id_of_neighbor_of_visited_node,
282  source_node_id,
283  distance_through_u);
284 
285  if (touced[id_of_neighbor_of_visited_node] == false) {
286  LOG_M(DEBUG3, "Inserting node " << id_of_neighbor_of_visited_node <<
287  " to heap with distance " << distance_through_u);
288  heap.insert(RankData(id_of_neighbor_of_visited_node, rankOfSourceNode, distance_through_u));
289  } else {
290  LOG_M(DEBUG3, "Decreasing distance of node " << id_of_neighbor_of_visited_node <<
291  " from " << min_distance[id_of_neighbor_of_visited_node] <<
292  " to " << distance_through_u);
293  heap.erase(RankData(id_of_neighbor_of_visited_node, (*ranking)[id_of_neighbor_of_visited_node], min_distance[id_of_neighbor_of_visited_node]));
294  heap.insert(RankData(id_of_neighbor_of_visited_node, rankOfSourceNode, distance_through_u));
295  }
296  LOG_M(DEBUG5, "Updating rank of node " << id_of_neighbor_of_visited_node <<
297  " rank=" << rankOfSourceNode <<
298  " old rank=" << (*ranking)[id_of_neighbor_of_visited_node]);
299  touced[id_of_neighbor_of_visited_node] = true;
300  min_distance[id_of_neighbor_of_visited_node] = distance_through_u;
301  (*ranking)[id_of_neighbor_of_visited_node] = rankOfSourceNode;
302  }
303  }
